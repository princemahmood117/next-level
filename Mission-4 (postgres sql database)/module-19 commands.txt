-- GROUP BY
select country, avg(age) from students group by country;  -- protita country er ekta group hobe, jekhane bangladesh er ekta group e jotogula 'bangladesh' ache ta joma hobe and pore others operation hobe (jemon avg(age) = total bangladesh er students er average age)

-- count students from each country
select country, count(*) as "total_students of this country" from students group by country;

-- count students by their grades
select grade, count(*) from students group by grade;


-- find those courses which has students more than 2 
select course, count(*) as "student amount" from students group by course having count(*) > 2;


-- find those countries which has average age of students more than 21
select country, avg(age) as "average age" from students group by country having avg(age) > 22;
